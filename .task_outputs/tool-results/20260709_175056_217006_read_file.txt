def _assistant_to_message(message) -> dict:
    record = {
        "role": "assistant",
        "content": message.content or "",
    }
    if message.tool_calls:
        record["tool_calls"] = [
            tool_call.model_dump() if hasattr(tool_call, "model_dump") else tool_call
            for tool_call in message.tool_calls
        ]
    return record


def _get_tool_name_and_args(tool_call) -> Tuple[str, dict]:
    if isinstance(tool_call, dict):
        function = tool_call.get("function", {})
        name = function.get("name", "")
        raw_args = function.get("arguments") or "{}"
    else:
        name = tool_call.function.name
        raw_args = tool_call.function.arguments or "{}"
    if isinstance(raw_args, str):
        return name, json.loads(raw_args or "{}")
    return name, raw_args


def _tool_call_id(tool_call) -> str:
    if isinstance(tool_call, dict):
        return tool_call.get("id", "")
    return getattr(tool_call, "id", "")


def _tool_name(tool_call) -> str:
    if isinstance(tool_call, dict):
        return tool_call.get("function", {}).get("name", "")
    return tool_call.function.name


def _dispatch_teammate_tool(name: str, args: dict, member_name: str, team_manager) -> str:
    if name == "bash":
        return run_bash(args["command"])
    if name == "read_file":
        return read_file(args["path"], args.get("limit"))
    if name == "write_file":
        return write_file(args["path"], args["content"])
    if name == "edit_file":
        return edit_file(args["path"], args["old_text"], args["new_text"])
    if name == "team_send_message":
        return team_manager.send_from_teammate(
            sender=member_name,
            to=args.get("to", team_manager.lead_name),
            content=args["content"],
            msg_type=args.get("msg_type", "message"),
        )
    if name == "team_check_inbox":
        return team_manager.check_inbox(agent=member_name)
    if name == "team_submit_plan":
        return team_manager.submit_plan(
            sender=member_name,
            plan=args["plan"],
            risk_level=args.get("risk_level", ""),
            target_files=args.get("target_files") or [],
        )
    if name == "team_protocol_status":
        return team_manager.protocol_status(
            request_id=args.get("request_id", ""),
            include_resolved=args.get("include_resolved", True),
        )
    if name == "task_list":
        return TASK_MANAGER.list_all(
            include_completed=args.get("include_completed", True),
            owner=args.get("owner"),
            status=args.get("status"),
        )
    if name == "task_get":
        return TASK_MANAGER.get(args["task_id"])
    if name == "task_complete":
        owner = args.get("owner") or member_name
        result = TASK_MANAGER.complete(
            task_id=args["task_id"],
            owner=owner,
            summary=args.get("summary", ""),
        )
        if result.startswith("已完成") and team_config.get("AUTONOMOUS_NOTIFY_LEAD_ON_COMPLETE", True):
            team_manager.send_structured_message(
                sender=member_name,
                to=team_manager.lead_name,
                content=result,
                msg_type="result",
                metadata={"task_id": args["task_id"], "event": "task_completed"},
            )
            AUTONOMOUS_MANAGER.append_event(
                "complete",
                member_name,
                {"task_id": args["task_id"], "message": result},
            )
        return result
    return f"Error: unknown teammate tool {name}"


def _max_output_tokens() -> int:
    return int(compact_config.get("MAX_OUTPUT_TOKENS", 2048))


def _clip(text: str, max_chars: int) -> str:
    if len(text) <= max_chars:
        return text
    return text[:max_chars] + "\n\n...[teammate 结果过长，已截断]"


def _format_inbox_messages(messages) -> str:
    return "\n".join(f"From {message.sender} [{message.type}]: {message.content}" for message in messages)


def _read_teammate_inbox(member_name: str, state: LoopState, team_manager) -> str:
    messages = team_manager.bus.read_inbox(
        member_name,
        consume=True,
        limit=int(team_config.get("MAX_INBOX_MESSAGES_PER_TURN", 20)),
    )
    if not messages:
        return "continue"
    action, remaining = handle_teammate_protocol_messages(member_name, messages, team_manager)
    if remaining:
        state.messages.append({
            "role": "user",
            "content": f"<team_inbox>\n{_format_inbox_messages(remaining)}\n</team_inbox>",
        })
    return action


def _idle_until_message(member: TeamMemberRecord, state: LoopState, team_manager) -> str:
    team_manager.record_member_status(member.name, "idle", turn_count=state.turn_count)
    if team_config.get("TEAM_IDLE_NOTIFY_LEAD", True):
        team_manager.send_structured_message(
            sender=member.name,
            to=team_manager.lead_name,
            content=f"teammate {member.name} 当前空闲，等待新消息或关闭请求。",
            msg_type="idle_notification",
            metadata={"status": "idle"},
        )

    poll_seconds = float(team_config.get("AUTONOMOUS_IDLE_SCAN_SECONDS", team_config.get("TEAM_IDLE_POLL_SECONDS", 1)))
    max_idle_seconds = float(team_config.get("AUTONOMOUS_IDLE_TIMEOUT_SECONDS", team_config.get("TEAM_IDLE_MAX_SECONDS", 0)) or 0)
    started = time.time()
    while True:
        action = _read_teammate_inbox(member.name, state, team_manager)
        if action == "shutdown":
            return "shutdown"
        if state.messages and state.messages[-1].get("content", "").startswith("<team_inbox>"):
            team_manager.record_member_status(member.name, "running", turn_count=state.turn_count)
            return "continue"

        claim_result = AUTONOMOUS_MANAGER.try_claim_next(member.name, role=member.role)
        if claim_result.claimed and claim_result.task:
            state.messages.append({
                "role": "user",
                "content": AUTONOMOUS_MANAGER.format_claimed_task(claim_result.task, member.name),
            })
            team_manager.record_member_status(member.name, "running", turn_count=state.turn_count)
            if team_config.get("AUTONOMOUS_NOTIFY_LEAD_ON_CLAIM", True):
                team_manager.send_structured_message(
                    sender=member.name,
                    to=team_manager.lead_name,
                    content=f"{member.name} 已自动认领 {claim_result.task.id}：{claim_result.task.subject}",
                    msg_type="message",
                    metadata={"task_id": claim_result.task.id, "event": "task_claimed"},
                )
            return "continue"
        if max_idle_seconds > 0 and time.time() - started >= max_idle_seconds:
            team_manager.record_member_status(member.name, "stopped", turn_count=state.turn_count)
            return "shutdown"
        time.sleep(poll_seconds)


def run_teammate_agent(
    member: TeamMemberRecord,
    client,
    model_id: str,
    hook_manager,
    workdir: str,
    team_manager,
) -> str:
    """运行一个 teammate 的独立 Agent 循环。"""
    max_turns = int(team_config.get("MAX_TEAMMATE_TURNS", 10))
    max_result_chars = int(team_config.get("TEAMMATE_RESULT_MAX_CHARS", 6000))
    allowed_tools = [tool["function"]["name"] for tool in TEAMMATE_AVAILABLE_TOOLS]
    state = LoopState(messages=[{
        "role": "user",
        "content": (
            f"Lead 分配给你的任务：\n{member.prompt}\n\n"
            f"你的角色：{member.role}\n"
            "完成后请调用 team_send_message(to=\"lead\", msg_type=\"result\", content=...) 汇报。"
        ),
    }])
    recovery_manager = RecoveryManager()
    runtime_context = ToolRuntimeContext(
        parent_messages=state.messages,
        client=client,
        model_id=model_id,
        hook_manager=hook_manager,
        workdir=workdir,
        subagent_depth=1,
        allowed_tools=allowed_tools,
        compact_manager=COMPACT_MANAGER,
    )

    while state.turn_count < max_turns:
        state.turn_count += 1
        team_manager.record_member_status(member.name, "running", turn_count=state.turn_count)
        if _read_teammate_inbox(member.name, state, team_manager) == "shutdown":
            return f"teammate {member.name} 已按 shutdown_request 退出。"

        active_model_id = recovery_manager.state.current_model_id or model_id
        hook_manager.run_hooks(
            "BeforeModelCall",
            HookContext(
                event="BeforeModelCall",
                messages=state.messages,
                model_id=active_model_id,
                metadata={
                    "turn_count": state.turn_count,
                    "client": client,
                    "compact_manager": COMPACT_MANAGER,
                    "subagent_depth": runtime_context.subagent_depth,
                    "team_member": member.name,
                    "allow_subagent_llm_compact": compact_config.get("ENABLE_SUBAGENT_LLM_COMPACT", False),
                },
            ),
        )

        def build_messages():
            return normalize_messages([
                {
                    "role": "system",
                    "content": build_teammate_system_prompt(
                        workdir,
                        runtime_context={
                            "teammate_name": member.name,
                            "teammate_role": member.role,
                            "team_rules": TEAMMATE_SYSTEM_RULES,
                        },
                    ),
                },
                *state.messages,
            ])

        response = recovery_manager.create_chat_completion(
            client=client,
            model_id=model_id,
            build_messages=build_messages,
            tools=TEAMMATE_AVAILABLE_TOOLS,
            tool_choice="auto",
            max_tokens=_max_output_tokens(),
            compact_manager=COMPACT_MANAGER,
            state_messages=state.messages,
        )
        assistant_message = response.choices[0].message
        state.messages.append(_assistant_to_message(assistant_message))

        continuation_message = recovery_manager.build_continuation_message(response.choices[0].finish_reason)
        if continuation_message:
            state.messages.append(continuation_message)
            continue

        tool_calls = assistant_message.tool_calls or []
        if not tool_calls:
            result = _clip(assistant_message.content or "", max_result_chars)
            if result.strip():
                team_manager.send_from_teammate(member.name, team_manager.lead_name, result, msg_type="result")
            idle_action = _idle_until_message(member, state, team_manager)
            if idle_action == "shutdown":
                return result or f"teammate {member.name} 已退出。"
            continue

        for tool_call in tool_calls:
            try:
                name, args = _get_tool_name_and_args(tool_call)
            except Exception as exc:
                logger.exception("[team] teammate 解析 tool_call 失败")
                result = f"Error: invalid teammate tool call: {exc}"
                state.messages.append({"role": "tool", "tool_call_id": _tool_call_id(tool_call), "content": result})
                continue

            if name not in allowed_tools:
                result = f"Error: teammate cannot use tool {name}"
            else:
                pre_result = hook_manager.run_hooks(
                    "PreToolUse",
                    HookContext(
                        event="PreToolUse",
                        tool_name=name,
                        tool_input=args,
                        metadata={"tool_call_id": _tool_call_id(tool_call), "team_member": member.name},
                    ),
                )
                if pre_result.blocked:
                    reason = pre_result.block_reason or "blocked by hook"
                    result = reason if reason.startswith("Permission denied:") else f"Tool blocked by hook: {reason}"
                else:
                    try:
                        result = _dispatch_teammate_tool(name, args, member.name, team_manager)
                    except Exception as exc:
                        logger.exception("[team] teammate 工具执行异常: %s", name)
                        result = f"Error while running teammate tool {name}: {exc}"
                    post_result = hook_manager.run_hooks(
                        "PostToolUse",
                        HookContext(
                            event="PostToolUse",
                            tool_name=name,
                            tool_input=args,
                            tool_output=result,
                            metadata={"tool_call_id": _tool_call_id(tool_call), "team_member": member.name},
                        ),
                    )
                    if post_result.updated_output is not None:
                        result = post_result.updated_output
            state.messages.append({
                "role": "tool",
                "tool_call_id": _tool_call_id(tool_call),
                "content": result,
            })

    result = f"teammate {member.name} 超过最大轮数 {max_turns}，已停止。"
    team_manager.send_from_teammate(member.name, team_manager.lead_name, result, msg_type="error")
    team_manager.record_member_status(member.name, "failed", turn_count=state.turn_count, error=result)
    return result
